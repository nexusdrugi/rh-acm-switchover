# NAME
acm-switchover-rbac - Validate RBAC permissions for ACM Switchover

# SYNOPSIS
**acm-switchover-rbac** [OPTIONS]

# DESCRIPTION
Checks the required Kubernetes RBAC permissions to ensure switchover can run.

# OPTIONS
--context, --primary-context, --secondary-context, --include-decommission, --skip-observability, --version

acm-switchover(1), acm-switchover-state(1)
